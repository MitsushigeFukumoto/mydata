\begin{figure}[h]
    \begin{center}
        \includegraphics[width=0.9\textwidth]{AIT_logo.png}
        \caption{図のキャプションは下}
        % labelはcaptionの直後に書く
        \label{fig:sample_fig}
    \end{center}
\end{figure}
\end{lstlisting}
