tkinterでファイルオープンアプリ

ファイルダイアログを開いて，ファイルやフォルダを選択します．

fileio_tkinter.python

import os
import tkinter as tk
import tkinter.filedialog
import tkinter.messagebox

class TkinterFileIO:
    def __init__(self):
        root = tk.Tk()
        root.title('第５回ファイルIO')
        root.geometry("500x350")

        button = tk.Button(root, text='ファイルダイアログを開く', font=('', 20),
                           width=24, height=1, bg='#999999', activebackground="#aaaaaa")
        button.bind('<ButtonPress>', self.file_dialog)
        button.pack(pady=40)

        self.file_name = tk.StringVar()
        self.file_name.set('未選択です')
        label = tk.Label(textvariable=self.file_name, font=('', 12))
        label.pack(pady=0)

        button = tk.Button(root, text='フォルダダイアログを開く', font=('', 20),
                           width=24, height=1, bg='#999999', activebackground="#aaaaaa")
        button.bind('<ButtonPress>', self.folder_dialog)
        button.pack(pady=40)

        self.folder_name = tk.StringVar()
        self.folder_name.set('未選択です')
        label = tk.Label(textvariable=self.folder_name, font=('', 12))
        label.pack(pady=10)

        root.mainloop()

    def file_dialog(self, event):
        fTyp = [("All Files", "*.*"),("Python Files", "*.py")]
        iDir = os.path.abspath(os.path.dirname(__file__))
        #iDir = os.getcwd()
        file_name = tk.filedialog.askopenfilename(filetypes=fTyp, initialdir=iDir)
        if len(file_name) == 0:
            self.file_name.set('選択をキャンセルしました')
        else:
            tkinter.messagebox.showinfo('TkinterFileIO',f'{file_name} が選択されました')
            self.file_name.set(file_name)

    def folder_dialog(self, event):
        #iDir = os.path.abspath(os.path.dirname(__file__))
        iDir = os.getcwd()
        folder_name = tk.filedialog.askdirectory(initialdir=iDir)
        if len(folder_name) == 0:
            self.folder_name.set('選択をキャンセルしました')
        else:
            tkinter.messagebox.showinfo('TkinterFileIO',f'{folder_name} が選択されました')
            self.folder_name.set(folder_name)


if __name__ == '__main__':
    TkinterFileIO()

簡単な解説

  1. tkinter.filedialog.* を呼び出すことで，ファイルの選択，ファイルの読み込み，書き込みなどファイル操作ができる

  2. tkinter.filedialog.askopenfilenameの引数にはどの多種類のファイルを開くか，どのディレクトリを最初に表示するかなどを指定できる

  • https://docs.python.org/ja/3/library/dialog.html?highlight=filedialog#tkinter.filedialog.askopenfilename

  • ファイルの種類を指定する場合，タブルを配列に追加していく fTyp.append(("CSV Files", "*.csv"))

  • ファイルやディレクトリを探す初期ディレクトリは，プログラムファイルの置かれている場所(os.path.abspath(os.path.dirname(__file__)))やプログラムを実行している場所(os.getcwd())を指定する．

  1. TkinterFileIOクラスには，ボタンクリック時の動作が関数として定義されている
  • クラス内で変数を定義する時は，self.var_name (i.e. self.file_name = tk.StringVar())
  1. tkinter.messageboxを利用するとポップアップを表示させることができる
