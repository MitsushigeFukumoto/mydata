from tkinter import *
from tkinter import ttk

def show_selection():
    for i in lb.curselection():
        print(lb.get(i))
    
if __name__ == '__main__':
    root = Tk()
    root.title('Scrollbar 2')
    root.columnconfigure(0, weight=1)
    root.rowconfigure(0, weight=1)

    # Frame
    # tkkのFrameを使っていることに注目
    frame = ttk.Frame(root, padding=10)
    frame.grid(sticky=(E, W, S, N))
    # weightは縦と横の長さの割合：同じ値だと同じ比率で子が配置される
    # frameはroot全体に置きたいので場所(0, 0)でweightは両方１
    frame.columnconfigure(0, weight=1)
    frame.rowconfigure(0, weight=1)

    # Listbox
    currencies = ('JPY', 'USD', 'EUR', 'CNY', 'MXN', 'CAD')
    v1 = StringVar(value=currencies)
    lb = Listbox(frame, listvariable=v1, height=3)
    # column=0(上)
    lb.grid(row=0, column=0, sticky=(E, W, S, N))

    # Scrollbar
    scrollbar = ttk.Scrollbar(
        frame,
        orient=VERTICAL,
        command=lb.yview)
    lb['yscrollcommand'] = scrollbar.set
    # stickyに北（上）と南（下）のみ指定する
    # column=1(真ん中)
    scrollbar.grid(row=0, column=1, sticky=(N, S))

    # Button
    button1 = ttk.Button(
        frame, text='OK',
        command=lambda: show_selection())
    button1.grid(
        row=1, column=0, columnspan=2,
        pady=5, sticky=(S))

    root.mainloop()

まとめ

ウィンドウのサイズ変更に追従するために，Wigetを並べるgridを使う．(row, column, columnspanなど)

stickyには，ペイン領域の中に子ウィンドウがどう置かれるかを指定する．指定する値は “n”, “s”, “e”, “w” からなる 0 文字以上の文字列． それぞれの文字は子ウィンドウが (北、南、東、西の) どの辺に対して追随するかに対応する．
