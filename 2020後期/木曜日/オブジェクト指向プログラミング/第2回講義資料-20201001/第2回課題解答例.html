#!/usr/bin/env python
# -*- coding: utf-8 -*-
import csv

def lecture02_01():

    # 武器データの読み込み
    weapon_header = []
    weapon_data = []
    with open('lecture02_Weapon.csv') as f:
        reader = csv.reader(f)
        for row in reader:
            if len(weapon_header) == 0:
                 weapon_header = row
            else:
                 weapon_data.append(row)

    # ヒーローデータの読み込み
    hero_header = []
    hero_data = []
    with open('lecture02_Hero.csv') as f:
        reader = csv.reader(f)
        for row in reader:
            if len(hero_header) == 0:
                hero_header = row
            else:
                hero_data.append(row)

    # HeroのデータにWeaponを装備(該当するステータスを足し合わせる)処理
    # ヒーローデータのWeaponと武器データのIDが等しことを条件にする
    for h in hero_data:
        for w in weapon_data:
            weapon_id = h[7] # インデックス7番目にweapon_id
            id = w[0] # インデックス0番目にid
            if weapon_id == id:
                # enumerateでインデックスと値を同時に取り出せる
                h[2:6] = [int(h[2+i]) + int(v) for i, v in enumerate(w[2:6])]

    # ヒーロデータの名前が太郎であれば、太郎のステータス情報を出力する
    for h in hero_data:
        if h[1] == "太郎":
            print("太郎のステータス")
            print(f'ID: {h[0]}')
            print(f'名前: {h[1]}')
            print(f'HP: {h[2]}')
            print(f'MP: {h[3]}')
            print(f'Atk: {h[4]}')
            print(f'Def: {h[5]}')
            print(f'年齢: {h[6]}')
            print(f'武器id: {h[7]}')

if __name__ == '__main__':
    lecture02_01()

#!/usr/bin/env python
# -*- coding: utf-8 -*-
import xml.etree.ElementTree as ET

def lecture02_02():
    tree = ET.parse('lecture02_data.xml')
    books = tree.getroot()

    #bookリスト
    book_tag = []
    book_attrib = []
    for book in books:
        book_tag.append(book.tag)
        book_attrib.append(book.attrib)

    #articleリスト
    article_tag = None
    article_attrib = []
    for article in books.find(book_tag[0]):
        article_tag = article.tag
        article_attrib.append(article.attrib)

    #novelリスト
    novel_tag = None
    novel_attrib = []
    for novel in books.find(book_tag[1]):
        novel_tag = novel.tag
        novel_attrib.append(novel.attrib)

    #辞書型へ返還
    book_layer = {article_tag: article_attrib, novel_tag: novel_attrib}
    h = {books.tag: book_layer}

    #出力
    print(h)
    # {'book': {'chapter': [{'id': '1', 'name': '1日のはじまり', 'pages': '2'}, {'id': '2', 'name': '朝食', 'pages': '8'}, {'id': '3', 'name': '仕事', 'pages': '6'}, {'id': '4', 'name': '帰宅後', 'pages': '2'}, {'id': '5', 'name': '新しい朝', 'pages': '1'}]}}

if __name__ == '__main__':
    lecture02_02()
