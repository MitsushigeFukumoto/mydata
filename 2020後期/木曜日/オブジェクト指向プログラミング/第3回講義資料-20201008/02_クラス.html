クラスの定義と継承

クラスを定義します．

この後の演習でこのクラスを使いますので，lecture03_hero.pyという名前でファイルに保存してください．

class Hero:

    # どのインスタンスでも共通の値が入る変数
    # 複数のHeroインスタンスのjob_statusは共通なのでHero職へのバフに使う
    # 例：魔法使いによる職業ステータスの向上はjob_status値を変更
    job_status = {'hp': 0}

    # コンストラクタ
    # デフォルト値を指定することもできる
    # なお，引数が異なるコンストラクタを複数定義することはできない
    def __init__(self, name=None, hp=0):
        self.name = name
        self.hp = hp

    # デストラクタ
    def __del__(self):
        message = '一時パーティー離脱' if self.hp>0 else '死亡のためパーティー離脱'
        print(f'{self.name}は{message}')

    # 特殊な関数の宣言
    def __str__(self):
        return f'"名前": "{self.name}", "HP": {self.hp + self.job_status["hp"]}'


if __name__ == '__main__':
    hero1 = Hero('愛知太郎') # Heroクラスのインスタンスを生成
    print(f'hero1={str(hero1)}') # 文字列に変換
    hero2 = Hero(name='愛知花子', hp=100) # 引数名を指定してコンストラクタを呼び出し
    print(f'hero2={str(hero2)}') # 文字列に変換
    hero1.job_status['hp'] += 20 # hero2のクラスインスタンスにもHP底上げバフがかかる
    print(f'Hero.job_status = {Hero.job_status}') # Heroクラスのjob_statusの値が変わる
    print(f'hero1={str(hero1)}') # 文字列に変換
    del hero1 # インスタンスを削除=>死亡のためパーティー離脱
    print(f'hero2={str(hero2)}') # 文字列に変換
    del hero2 # インスタンスを削除=>一時パーティー離脱
