        odict = xmltodict.parse(xml.read())

    # odictはOrderedDict型なので課題に従いdict型にする

    lecture02_02()

from readdata import *
import matplotlib.pyplot as plt
import numpy as np
import scipy.stats as scst

print('input data type...')
print('Data is set in the list RawData -> 1')
print('Data is set in the text file    -> 2')
print('Data is set in the csv  file    -> 3')
ck=input('\nYour choice=')
try:
    if ck=='1':
        RawData,f_out,f_out_nm=ReadData_lst()
    elif ck=='2':
        RawData,f_out,f_out_nm=ReadData_txt()
    elif ck=='3':
        RawData,f_out,f_out_nm=ReadData_csv()
    else:
        print('\nInvalid choice...')
        raise Exception()
        
    f_out.write('Data...\n')
    print('Data...')
    for v in RawData:
        f_out.write('{0}\n'.format(v))
        print(v)
        
    VarNames=RawData[0]
    
    ID=[]
    y=[]
    X=[]
    for v in RawData[1:]:
        ID.append(v[0])
        y.append([v[1]])
        X.append([1,v[2]])
    y=np.array(y)
    X=np.array(X)
    
    b=np.linalg.inv(X.transpose()@X)@X.transpose()@y
    print('b=',b)
    
    X1=X.transpose()[1]
    vy=[]
    for v in y:
        vy.append(v[0])
    y=vy
    r,p=scst.pearsonr(X1,y)
    print('r=',r)
    R2=r**2
    print('R2=',R2)
    
    min_x=np.amin(X1)
    max_x=np.amax(X1)
    y_left=b[0][0]+b[1][0]*min_x
    y_right=b[0][0]+b[1][0]*max_x
    plt.plot([min_x,max_x],[y_left,y_right],'b-')
    
    dis_x=(max_x-min_x)*0.01
    dis_y=(np.amax(y)-np.amin(y))*0.01
    
    plt.plot(X1,y,'bo')
    
    for vx,vy,name in zip(X1,y,ID):
        plt.text(vx+dis_x,vy+dis_y,name)
    plt.xlabel(VarNames[2])
    plt.ylabel(VarNames[1])
    plt.title('Simple Linear Regression\n'+'r={0:.3} $R^2$={1:.3}'.format(r,R2))
    
    plt.show()
    
    f_out.write('\nb0<Constant>={0:.5}\n'.format(b[0][0]))
    f_out.write('b1<{0}>={1:.5}\n'.format(VarNames[2],b[1][0]))
    f_out.write('\nr={0:.3} R^2={1:.3}\n'.format(r,R2))
    
    f_out.close()
    print('{0} was saved.'.format(f_out_nm))
    
except Exception as e:
    print('\nException...',e)

import csv

def ReadData_list():
    RawData=[['City','Temperature','Latitude'],
    ['Sapporo',15,43.1],
    ['Sendai',20,38.3],
    ['Niigata',17,37.9],

    ['Osaka',22,34.7],

    ['Kagoshima',23,31.6],

    file_name=input('Output file name=')
    f_out=open(file_name,'w')
    
    return RawData,f_out,file_name
    
def ReadData_txt():
    f_in_nm=input('input data file =')
    f_in=open(f_in_nm,'r')
    f_data=f_in.readlines()
    pos=0
    while Ture:
        if len(f_data[pos])>0:
            if f_data[pos][0]=='/':
                pos+=1
                break
        pos+=1
        
    RawData=[]
    ck=0
    while Ture:
        if f_data[pos][0]=='/':
            break
        t_str=f_data[pos].split()
        if ck==0:
            RawData.append(t_str)
            ck=1
        else:
            RawData.append([t_str[0],float(t_str[1]),float(t_str[2])])
        pos+=1
        
    file_name=input('Output file name =')
    f_out=open(file_name,'w')
    f_out.write('input data = {}\n'.format(f_in_nm))
    
    return RawData,f_out,file_name

def ReadData_csv():
    f_in_nm=input('input data file(*.csv) =')
    with open(f_in_nm,'r') as f:
        csv_data=[d for d in csv.reader(f)]
        
    RawData=[]
    for i in range(len(csv_data)):
        if i==0:
            RawData.append(csv_data[i])
        else:
            RawData.append([csv_data[i][0],float(csv_data[i][1]),float(csv_data[i][2])])
            
    file_name=input('Output file name=')
    f_out=open(file_name,'w')
    f_out.write('Input data = {}\n'.format(f_in_nm))
    
    return RawData,f_out,file_name
