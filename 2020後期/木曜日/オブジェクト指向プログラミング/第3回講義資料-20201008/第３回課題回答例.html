#!/usr/bin/env python
# -*- coding: utf-8 -*-
import csv

class MyCsvReader():
    # コンストラクタでCSVファイルを読み込み
    def __init__(self): 
        # 武器データの読み込み
        self.weapon_header = []
        self.weapon_data = []
        with open('lecture02_Weapon.csv') as f:
            reader = csv.reader(f)
            for row in reader:
                if len(self.weapon_header) == 0:
                     self.weapon_header = row
                else:
                     self.weapon_data.append(row)

        # ヒーローデータの読み込み
        self.hero_header = []
        self.hero_data = []
        with open('lecture02_Hero.csv') as f:
            reader = csv.reader(f)
            for row in reader:
                if len(self.hero_header) == 0:
                    self.hero_header = row
                else:
                    self.hero_data.append(row)
    
    def equip_weapon(self, weapon_id=None):
        if weapon_id:
            # 太郎のデータにWeaponを装備(該当するステータスを足し合わせる)処理
            # 太郎はインデックス0に入っているとする
            h = self.hero_data[0]
            for w in self.weapon_data:
                id = w[0]
                if weapon_id == (int)(id): # weapon_idに数字を渡した場合はintにキャストする 
                    # enumerateでインデックスと値を同時に取り出せる
                    h[2:7] = [int(h[2+i]) + int(v) for i, v in enumerate(w[2:7])]

            # 太郎のステータス情報を出力する
            print("太郎のステータス")
            print(f'ID: {h[0]}')
            print(f'名前: {h[1]}')
            print(f'HP: {h[2]}')
            print(f'MP: {h[3]}')
            print(f'Atk: {h[4]}')
            print(f'Def: {h[5]}')
            print(f'Age: {h[6]}')
            print(f'武器id: {h[7]}')

if __name__ == '__main__':
    my_csv = MyCsvReader()
    my_csv.equip_weapon(1)
