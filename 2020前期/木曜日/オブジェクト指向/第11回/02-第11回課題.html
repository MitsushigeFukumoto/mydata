1. ガチャの課金シミュレーションプログラム

サンプルプログラム(後述)の永続化を行います。

課金を行ったデータは、プログラムが終了するごとに消えてしまいます。 この課題では、その課金データをシリアライズし、data.datというファイル名で保存しておくことでシュミレーション上ではありますが過去の行いをデータで保存できるようにします。

サンプルソースは、単一の実行のみでデータの保存等は一切考慮されていません。 プログラム実行時に、data.datがあるかどうかを確認し、あれば読み込んで展開。なければ初期化を行うように修正します。

保存するデータのファイル名は固定で良いです。

最初から行いたい場合は、data.datを削除すると良いでしょう。

また、永続化には必ずSerializableを実装することで対応してください。

用意したサンプルプログラムには、TODO:のコメントにて修正するべき箇所が書かれていますが、それ以外にも修正しなければならないところは存在します。 プログラムソースをよく読み、どこをどう直すかよく考えましょう。

課金データクラス(ChargeData.java)

package jp.ac.ait.k00000;

import java.util.Random;

public class ChargeData {

  /**
   * ガチャ1回分の課金単価
   */
  private int unitPrice;
  /**
   * 課金額合計
   */
  private long chargedTotal;
  /**
   * SSRが出る確率(%) 小数点以下4桁まで対応
   */
  private double chanceOfWinning;
  /**
   * 挑戦回数
   */
  private long challengeCount;

  /**
   * SSRが出た回数
   */
  private int hitCount;

  /**
   * @return the unitPrice
   */
  public int getUnitPrice() {
    return unitPrice;
  }

  /**
   * @param unitPrice the unitPrice to set
   */
  public void setUnitPrice(int unitPrice) {
    this.unitPrice = unitPrice;
  }

  /**
   * @return the chargedTotal
   */
  public long getChargedTotal() {
    return chargedTotal;
  }

  /**
   * @param chargedTotal the chargedTotal to set
   */
  public void setChargedTotal(long chargedTotal) {
    this.chargedTotal = chargedTotal;
  }

  /**
   * @return the chanceOfWinning
   */
  public double getChanceOfWinning() {
    return chanceOfWinning;
  }

  /**
   * @param chanceOfWinning the chanceOfWinning to set
   */
  public void setChanceOfWinning(double chanceOfWinning) {
    this.chanceOfWinning = chanceOfWinning;
  }

  /**
   * @return the challengeCount
   */
  public long getChallengeCount() {
    return challengeCount;
  }

  /**
   * @param challengeCount the challengeCount to set
   */
  public void setChallengeCount(long challengeCount) {
    this.challengeCount = challengeCount;
  }

  /**
   * @return the hitCount
   */
  public int getHitCount() {
    return hitCount;
  }

  /**
   * @param hitCount the hitCount to set
   */
  public void setHitCount(int hitCount) {
    this.hitCount = hitCount;
  }

  /**
   *
   * @param unitPrice
   * @param chargedTotal
   * @param chanceOfWinning
   * @param challengeCount
   * @param hitCount
   */
  public ChargeData(int unitPrice, long chargedTotal, double chanceOfWinning, long challengeCount, int hitCount) {
    this.unitPrice = unitPrice;
    this.chargedTotal = chargedTotal;
    this.chanceOfWinning = chanceOfWinning;
    this.challengeCount = challengeCount;
    this.hitCount = hitCount;
  }

  /**
   * 引数なしの場合は、単価300、出現確率1%で初期化
   */
  public ChargeData() {
    this(300, 0, 1, 0, 0);
  }

  /**
   * 単価と確率だけ設定して初期化
   *
   * @param unitPrice
   * @param chanceOfWinning
   */
  public ChargeData(int unitPrice, double chanceOfWinning) {
    this(unitPrice, 0, chanceOfWinning, 0, 0);
  }

  /**
   * ガチャ回す 回した回数と、課金額合計が自動的に加算されます。
   *
   * @return 設定した確率で当選可否を返します
   */
  public boolean challenge() {

    // 課金回数と課金額を加算
    this.challengeCount += 1;
    this.chargedTotal += this.unitPrice;

    long rand = new Random().nextInt(100 * 10000);  //小数点以下4桁まで対応する
    if (rand < (this.chanceOfWinning * 10000)) {
      // 当選
      // SSR当選回数をインクリメント
      this.hitCount += 1;
      return true;
    } else {
      return false;
    }
  }

}

package jp.ac.ait.k00000;

import java.util.Random;
import java.util.Scanner;

public class ThrowMoneyAway {

  // 入力用Scanner
  private static final Scanner sc = new Scanner(System.in);

  public static void main(String[] args) {

    //TODO: "data.dat"のファイル名で保存されたデータがある場合、それを読み込んで以下の初期化は飛ばす
    //TODO: "data.dat"がない場合、もしくは読み込み失敗した場合、以下の初期化は行う
    ChargeData data = null;
    
    // ※↓初期化
    int price = 300;
    double ssr = 1.0;
    System.out.println("課金データが見つかりませんでした。初期設定を行います。");

    try {
      System.out.println("ガチャの基本データ:課金単価を整数値で入力してください。");
      price = Integer.parseInt(sc.nextLine());
      System.out.println("ガチャの基本データ:SSRの出現確率を実数値で入力してください。");
      ssr = Double.parseDouble(sc.nextLine());
    } catch (Exception e) {
      System.out.println("入力エラー");
      // 例外発生時は初期値とする(めんどうなのでここは何もしない)
    }
    // ガチャの基本データを設定して初期化
    data = new ChargeData(price, ssr);
    // ※↑ここまで初期化

    // ガチャデータの表示
    System.out.println("課金単価: " + data.getUnitPrice() + "円");
    System.out.println("SSRの確率: " + data.getChanceOfWinning() + "%");

    // SSRが出るか、諦めるまで無限ループ
    while (doChallenge(data)) {

      //TODO: "data.dat"に課金のデータを保存する

    }

    // ガチャへの無謀な挑戦の結果表示
    System.out.println("課金回数: " + data.getChallengeCount() + "回");
    System.out.println("課金額合計: " + data.getChargedTotal() + "円");
    System.out.println("SSR出現総数: " + data.getHitCount() + "回");

    //TODO: 当選していようがいまいが、"data.dat"に課金のデータを保存する

  }


  /**
   * ガチャ引くぞ！
   * @param data 課金データを渡す
   * @return trueで継続、falseで終了
   */
  private static boolean doChallenge(ChargeData data) {
    System.out.println("操作を選んで下さい。\n(1:課金してガチャを引く, それ以外:諦める)");

    int command = 0;
    try {
      command = Integer.parseInt(sc.nextLine());
    } catch (Exception e) {
      // 例外発生時は諦める選択とする
      command = 0;
    }

    if (command != 1) {
      System.out.println("「諦める」を選択しました。お疲れさまでした。");
      return false;
    }

    System.out.println("「課金してガチャを引く」を選択しました。");

    // 演出のための待ち時間
    for (int i = 0; i < 5; i++) {
      System.out.print("* ");
      try {
        Thread.sleep(300);
      } catch (InterruptedException e) {
        throw new RuntimeException(e);
      }
    }
    System.out.println();

    // ガチャを引く
    if (data.challenge()) {
      System.out.println("ガチャ結果: SSR の景品が出ました。");
      System.out.println("おめでとうございます！お疲れさまでした。");
      return false;
    }

    // ガチャの結果(SSRじゃなければ当たりじゃない)
    String[] rare = {"SR++", "SR+", "SR", "R++", "R+", "R", "N++", "N+", "N"};
    System.out.println("ガチャ結果: " + rare[new Random().nextInt(rare.length)]);

    // trueを返すことで挑戦継続
    return true;
  }

}

2.JPEG画像を壊すプログラム

画像ファイル形式であるJPEGデータは、ざっくりと言うと連続したデータの形式で圧縮を実現しています。そのため、バイナリデータの一部を1bitだけでも変更することで画像はどんどん壊れていきます。

ただ、連続データである関係上、ファイル自体の形式は保たれており、bitがずれたぐらいではJPEG画像としてプレビューは可能になっている可能性が高いです。

読み込む画像、壊して保存する画像ファイルのパスは、メインメソッド内で指定してください。プログラム中で固定値で良いです。また、バイト配列のどこかを操作する処理は、1bitだけの違いではほとんどわかりません。なので、何度破壊を行うかも指定できるようにしましょう。

3.ZIP圧縮するプログラム

この課題は、今回の内容だけでなく、前回の内容やStreamをいかに扱うかを問う総合的な課題です。

単一ファイル をzip形式で圧縮し、元のファイルと同じフォルダに、同じファイル名.zipとするメソッドを持つクラスを作成してください。
sample.csvだったら、sample.csv.zipで良いです。

注意点は以下の通りです。

  1. Zip圧縮を行う機能は、メソッドで独立させてください。
    • メソッド名、引数、戻り値の指定は各自にお任せしますが、汎用的に使用できるメソッドを作成してください。
  2. mainメソッドには動作確認用のプログラムを書いてください。
    • ファイルパスを画面から入力させるでも良いし、デバッグ用に用意したファイルを固定値で指定するのでも良いです。
  3. 複数ファイルができるなら複数ファイルができてもOKです。
  4. ディレクトリ指定での圧縮はできなくてよいですが、もし実装するならファイル指定、ディレクトリ指定どちらの指定でも正しく圧縮できるように実装してください。

ヒント

  • java.util.zipパッケージと、java.util.zip.ZipOutputStreamあたりを使用すればできるんじゃないかな。
    • java.util.zip.ZipOutputStreamには、java.util.zip.ZipEntryのインスタンスを受け取るputNextEntry(entry)があるよ。圧縮したいファイルをエントリーして登録していけばZipファイル作られる便利なライブラリだよ！
  • 指定されたファイルがない場合、圧縮ファイルが保存できない場合などのチェック処理を忘れずに⼊れよう。
  • ネットにあるプログラムを使ってみるのもだめとは言いません。
    • コピペで終わり、というのもできなくはないと思いますが、try-with-resouses構文や、NIOなどを使ってないややこしいプログラムをコピペしてませんか？
    • きちんと確認して提出してください。
