//import java.nio.file.Path;
//import java.nio.file.Paths;

// 以下はJava11より実装されたPath.ofを使う方法

Pathオブジェクトはファイル処理を行う上で真っ先に生成しなければならないことが多いため、変数に退避させておくことも考えると良いでしょう。

java.nio.file.Pathとjava.io.Fileの相互変換

古いAPIを使いたい場合など、nioパッケージではなく、java.io.Fileが必要になる場合もあります。

//import java.io.File;
//import java.nio.file.Path;

Path path = Path.of("path/to/file.txt");
// Path -> File
File file = path.toFile();
// File -> Path
Path pathFromFile = file.toPath();

ファイル存在確認, 読み取り可否確認

PC上にある通常のファイルなら以下の手法でファイルの存在確認、読み取り可否が可能です。

//import java.nio.file.Files;
//import java.nio.file.Path;
String filepath = "path/to/file.txt";

// ファイルの存在チェック
if (Files.exists(Path.of(filepath))) {
  System.out.println("ファイルあります");
} else {
  System.out.println("ファイルありません");
}

// ファイルの読み取り可否チェック
if (Files.isReadable(Path.of(filepath))) {
  System.out.println("ファイル読み込めます");
} else {
  System.out.println("ファイル読み込めません");
}

テキストファイルの文字コードを指定して読み込み

文字コードを指定する場合、Files.newBufferedReaderにて第2引数で指定します。

以下はShift-JISの文字コードのファイルを読み込みます。

//import java.io.IOException;
//import java.io.UncheckedIOException;
//import java.nio.file.Files;
//import java.nio.file.Path;
//import java.util.Scanner;
//import java.nio.charset.Charset;

try (Scanner sc = new Scanner(Files.newBufferedReader(Path.of("path/to/file.txt"), Charset.forName("Shift-JIS")))) {
    while (sc.hasNextLine()) {
        System.out.println(sc.nextLine());
    }
} catch (IOException e) {
    throw new UncheckedIOException(e);
}

古い日本語テキストファイルには、Shift-JISという文字コードが使用されていることがあり、これを使うことで文字化けを回避することができます。

また、Charset.defaultCharset()を使用するとシステム標準文字コードを指定でき、java.nio.charset.StandardCharsetsには、標準的に使用される文字コード数種類の定数が用意されています。

ここで紹介している読み込みメソッドには、このコードと同様に文字コードを指定して開くことができるものもあります。必要な場合は各自調べて使いましょう。

1行ずつファイル書き込み

書き込み時のオプションは、UTF-8(システムデフォルト)、新規作成、あれば消去してから作成、を指定しています。

//import java.io.BufferedWriter;
//import java.io.IOException;
//import java.io.UncheckedIOException;
//import java.nio.charset.Charset;
//import java.nio.file.Files;
//import java.nio.file.Path;

try (BufferedWriter bw = Files.newBufferedWriter(Path.of("path/to/file.txt"), Charset.defaultCharset())) {
  for (String line : lines) {
    bw.write(line);
    bw.newLine();
  }
} catch (IOException e) {
  throw new UncheckedIOException(e);
}

テキスト追記

Files.newBufferedWriterの第3引数以降にStandardOpenOptionを指定することで書き込み時の様々な設定を追加することが可能です。

//import java.io.BufferedWriter;
//import java.io.IOException;
//import java.io.UncheckedIOException;
//import java.nio.charset.Charset;
//import java.nio.file.Files;
//import java.nio.file.Path;
//import java.nio.file.StandardOpenOption;

try (BufferedWriter bw = Files.newBufferedWriter(Path.of("path/to/file.txt"), Charset.defaultCharset(), StandardOpenOption.APPEND)) {
  for (String line : lines) {
    bw.write(line);
    bw.newLine();
  }
} catch (IOException e) {
  throw new UncheckedIOException(e);
}
